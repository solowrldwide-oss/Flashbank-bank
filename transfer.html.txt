<script>
function checkLogin() {
  if (sessionStorage.getItem("loggedIn") !== "true") {
    window.location.href = "index.html";
  }
}

function lookup() {
  const acc = document.getElementById("acc").value;
  const amount = document.getElementById("amount").value;

  if (!acc || !amount) {
    document.getElementById("status").innerText = "Fill all fields first.";
    return;
  }

  document.getElementById("status").innerText = "Verifying account...";

  setTimeout(() => {
    document.getElementById("status").innerText = "Account verified ✔ Processing transfer...";

    setTimeout(() => {
      showReceipt(acc, amount);
    }, 1500);

  }, 1500);
}

function showReceipt(acc, amount) {
  document.getElementById("receipt").style.display = "block";

  document.getElementById("receiptText").innerText =
    `You sent $${amount} to account ${acc}. Transaction ID: TXN${Math.floor(Math.random()*99999999)}`;
}

function goBack() {
  window.location.href = "dashboard.html";
}
</script>
